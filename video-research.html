function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();

    if (sheet.getLastRow() === 0) {
      sheet.appendRow(['元動画URL','動画の内容','必要な素材','メモ','フォーマット','再現スコア','再現条件','撮影済み','登録日']);
      sheet.getRange(1,1,1,9).setFontWeight('bold').setBackground('#222222').setFontColor('#ffffff');
    }

    sheet.appendRow([
      data.url, data.content, data.materials, data.memo,
      data.format, data.score, data.checks, data.done, data.addedAt
    ]);

    return ContentService.createTextOutput('OK').setMimeType(ContentService.MimeType.TEXT);
  } catch(err) {
    return ContentService.createTextOutput('ERROR: ' + err).setMimeType(ContentService.MimeType.TEXT);
  }
}
